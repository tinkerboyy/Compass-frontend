angular.module('resourceAllocation.directive.agGridPagination',[
    'resourceAllocation.service.ResourceAllocationService'
])
    .controller("agPaginationController", ['$scope', 'ResourceAllocationService', function($scope, ResourceAllocationService) {
        var agPaginationController = this;
 
        agPaginationController.totalResults = 0;
        agPaginationController.previousEnabled = true;
        agPaginationController.nextEnabled = true;
        agPaginationController.startRow = 0;
        agPaginationController.endRow = 0;
        agPaginationController.currentPage = 0;
        agPaginationController.pageCount = 0;
 
        agPaginationController.goFirst = function() {
            var oldParams = angular.copy($scope.datasource.lastParams);
            var params = {
                startRow: 0,
                sortModel: oldParams.sortModel
            };
            $scope.datasource.source.getRows(params);
        };
 
        agPaginationController.goPrevious = function() {
            var oldParams = angular.copy($scope.datasource.lastParams);
            var startRow = oldParams.startRow - $scope.datasource.source.pageSize;
            var params = {
                startRow: startRow,
                sortModel: oldParams.sortModel
            };
            $scope.datasource.source.getRows(params);
        };
 
        agPaginationController.goNext = function() {
            var oldParams = angular.copy($scope.datasource.lastParams);
            var params = {
                startRow: oldParams.endRow,
                sortModel: oldParams.sortModel
            };
            $scope.datasource.source.getRows(params);
        };
 
        agPaginationController.goLast = function() {
            var oldParams = angular.copy($scope.datasource.lastParams);
            var totalRows = agPaginationController.totalResults;
            var pageSize = $scope.datasource.source.pageSize;
 
            var startRow = (totalRows % pageSize > 0) ? Math.floor(totalRows/pageSize) * pageSize : totalRows - pageSize;
            var params = {
                startRow: startRow,
                sortModel: oldParams.sortModel
            };
 
            $scope.datasource.source.getRows(params);
        };
 
        $scope.$watch(function() {
            return $scope.results;
        }, function(data) {
            var results = data;
 
            if (!data) {
                agPaginationController.resultCount = 0;
                agPaginationController.previousEnabled = false;
                agPaginationController.nextEnabled = false;
                return;
            }
            var reqObj = $scope.requestParams;
 
            agPaginationController.totalResults = data.total;
            agPaginationController.startRow = (reqObj.page * reqObj.limit) - reqObj.limit + 1;
            agPaginationController.endRow = Math.min(agPaginationController.startRow + reqObj.limit - 1, data.total);
            agPaginationController.currentPage = reqObj.page;
            agPaginationController.pageCount = Math.ceil(data.total/reqObj.limit);
 
            agPaginationController.previousEnabled = reqObj.page > 1;
            agPaginationController.nextEnabled = (reqObj.page < agPaginationController.pageCount);
        }, true);
    }])
 
    .directive('agGridPagination', [function() {
        return {
            restrict: "A",
            scope: {
                datasource: "=agGridPagination",
                results: "=",
                requestParams: "="
            },
            templateUrl: "resourceAllocation/directives/agGridPagination.tpl.html",
            controller: 'agPaginationController as paginationController'
        };
    }]);
