'use strict';
 
angular.module('scheduleSummary.directive.ScheduleAssociates', [
    'scheduleSummary.service.ScheduleAssociatesServices',
    'resourceAllocation.directive.CostCenterTypeahead',
    'scheduleSummary.directive.SelectDropdown',
    'scheduleSummary.directive.SchedulerGroupedPositions'
])
    .controller('ScheduleAssociatesController', ['$scope', 'ScheduleAssociatesService', '$filter',
        function ($scope, ScheduleAssociatesService, $filter) {
            var scheduleAssociatesCtrl = this;
            scheduleAssociatesCtrl.selectedCostCenter = null;
            scheduleAssociatesCtrl.selectedPosition = null;
            scheduleAssociatesCtrl.associates = [];
            scheduleAssociatesCtrl.groupedAssociates = [];
            scheduleAssociatesCtrl.isCollapsed = false;
 
            var userId = (typeof userId === 'undefined') ? $scope.userId : userId || 3;
            var deptId = (typeof deptId === 'undefined') ? $scope.deptId : deptId || '41760';
            var newTeamId = (typeof newTeamId === 'undefined') ? $scope.newTeamId : newTeamId || 15657;
            var weekEndingDate = (typeof weekEndingDate === 'undefined') ? $scope.weekEndingDate : $filter('date')(new Date(), 'MM/dd/yyyy');
 
            function showAssociates(userId, deptId, newTeamId, weekEndingDate) {
                ScheduleAssociatesService.getAllAssociates(userId, deptId, newTeamId, weekEndingDate)
                    .then(function (response) {
                        scheduleAssociatesCtrl.associates = response.associates;
                        scheduleAssociatesCtrl.groupedAssociates = _.groupBy(scheduleAssociatesCtrl.associates, 'jobDesc');
                    });
            }
            $scope.$watch('scheduleAssociatesCtrl.selectedCostCenter', function() {
                if(scheduleAssociatesCtrl.selectedCostCenter !== null) {
                    scheduleAssociatesCtrl.newCostCenter = angular.copy(scheduleAssociatesCtrl.selectedCostCenter);
                    newTeamId = scheduleAssociatesCtrl.newCostCenter.id;
                    showAssociates(userId, deptId, newTeamId, weekEndingDate);
                }
            });
 
            showAssociates(userId, deptId, newTeamId, weekEndingDate);
        }]
)
    .directive('scheduleAssociates', function() {
        return {
            restrict: 'E',
            scope: {
                userId: '=userId',
                deptId: '=deptId',
                newTeamId: '=newTeamId',
                weekEndingDate: '=date'
            },
            controller: 'ScheduleAssociatesController as scheduleAssociatesCtrl',
            templateUrl: 'scheduleSummary/directives/scheduleAssociates.tpl.html'
        };
 
    })
    .filter('searchPosition', [function() {
        return function(arr, selectedPosition) {
             if(!selectedPosition) {
                 return arr;
             }
 
            var result = {};
 
            if(selectedPosition.toUpperCase().indexOf(arr)) {
                result = _.pick(arr, selectedPosition);
            } else {
                throw 'Selected Position not in the group';
            }
 
            return result;
        };
    }]);
