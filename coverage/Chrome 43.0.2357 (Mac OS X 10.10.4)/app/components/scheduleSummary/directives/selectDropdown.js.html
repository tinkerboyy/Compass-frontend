'use strict';
 
angular.module('scheduleSummary.directive.SelectDropdown', [])
    .controller('SelectDropdownController', ['$scope', '$window',
        function ($scope, $window) {
            var selectDropdownCtrl = this;
            selectDropdownCtrl.dropdownList = [];
            selectDropdownCtrl.showDropDownList = false;
 
            selectDropdownCtrl.lookupsPending = 0;
 
            $scope.$window = $window;
 
            $scope.$watch($scope.responseData, function() {
                selectDropdownCtrl.dropdownList = $scope.responseData;
            });
 
            selectDropdownCtrl.selectPosition = function(searchposition) {
 
                selectDropdownCtrl.toggleDropDown(true);
                console.log(selectDropdownCtrl.searchPosition);
            };
 
            selectDropdownCtrl.selected = function(position, group) {
                if(position) {
                    selectDropdownCtrl.searchPosition = position;
                    $scope.selectedPosition = position;
                }
                selectDropdownCtrl.toggleDropDown();
                $scope.result({group: position});
            };
 
            selectDropdownCtrl.toggleDropDown = function(manualSetting) {
                selectDropdownCtrl.showDropDownList = manualSetting ? manualSetting : !selectDropdownCtrl.showDropDownList;
                selectDropdownCtrl.dropdownList = $scope.responseData;
                if (selectDropdownCtrl.showDropDownList) {
                    $scope.$window.onclick = function (event) {
                        closeSearchWhenClickingElsewhere(event, selectDropdownCtrl.toggleDropDown);
                    };
                } else {
                    selectDropdownCtrl.showDropDownList = false;
                    $scope.$window.onclick = null;
                }
            };
 
            function closeSearchWhenClickingElsewhere(event, callbackOnClose) {
                var clickedElement = event.target;
                if (!clickedElement) {
                    return;
                }
 
                var clickedOnTypeAheadHTML = checkParentChain(clickedElement);
 
                if (!clickedOnTypeAheadHTML) {
                    callbackOnClose();
                    return;
                }
            }
 
            function checkParentChain(element) {
                if (!element.parentElement) {
                    return false;
                }
 
                if (element.parentElement.classList.contains('selectDropdown')) {
                    return true;
                }
                else {
                    return checkParentChain(element.parentElement);
                }
            }
 
        }]
)
 
    .directive('selectDropdown', [
        function() {
            return {
                restrict: 'EA',
                scope: {
                    selectedPosition: '=ngModel',
                    responseData: '=data',
                    result: '&',
                    searchVal: '='
                },
                require: 'ngModel',
                templateUrl: 'scheduleSummary/directives/selectDropDown.tpl.html',
                controller: 'SelectDropdownController as selectDropdownCtrl',
                link: function(scope, elem, attr, ctrl) {
                    ctrl.$parsers.unshift(function(value) {
                        var valid = false;
 
                        if (value) {
                            valid = (value && value.trim().length > 0);
                            ctrl.$setValidity('invalidCostCenter', valid);
                        }
 
                        return valid ? value : undefined;
                    });
                }
            };
        }
    ]);
